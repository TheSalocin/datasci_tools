[docs]def json_from_file(filepath,multiple_objs = True):
    if multiple_objs:
        objs = []
        with open(filepath) as f:
            for jsonObj in f:
                studentDict = json.loads(jsonObj)
                objs.append(studentDict)
        return objs
    else:
        with open(filepath) as f:
            studentDict = json.loads(f)
        return studentDict
